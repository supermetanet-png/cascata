
-- Tabela para Webhooks de Entrada (Recebimento de Eventos Externos)
CREATE TABLE IF NOT EXISTS system.ingress_hooks (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    project_slug TEXT NOT NULL REFERENCES system.projects(slug) ON DELETE CASCADE,
    name TEXT NOT NULL,
    route_slug TEXT NOT NULL, -- A URL será /api/data/:project/hooks/in/:route_slug
    method TEXT DEFAULT 'POST',
    
    -- Configuração de Segurança (The Fortress)
    is_active BOOLEAN DEFAULT true,
    security_config JSONB DEFAULT '{
        "verify_signature": false,
        "header_key": "", 
        "secret": "", 
        "algorithm": "sha256",
        "allowed_ips": []
    }'::jsonb,

    -- Definição do Fluxo (No-Code Logic)
    flow_definition JSONB DEFAULT '[]'::jsonb,
    
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    
    -- Garante que o slug da rota seja único dentro do projeto
    UNIQUE(project_slug, route_slug)
);

-- Logs específicos para Ingress (Para debug e auditoria de chamadas externas)
CREATE TABLE IF NOT EXISTS system.ingress_logs (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    hook_id UUID REFERENCES system.ingress_hooks(id) ON DELETE CASCADE,
    project_slug TEXT NOT NULL,
    status_code INTEGER,
    client_ip TEXT,
    payload JSONB, -- O que chegou
    response_body JSONB, -- O que respondemos ou erro
    duration_ms INTEGER,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX IF NOT EXISTS idx_ingress_logs_hook ON system.ingress_logs (hook_id, created_at DESC);
