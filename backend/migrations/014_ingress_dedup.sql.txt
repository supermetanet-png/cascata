
-- Adiciona coluna para rastrear ID único do evento (Idempotência)
ALTER TABLE system.ingress_logs 
ADD COLUMN IF NOT EXISTS event_id TEXT;

-- Índice para verificação ultra-rápida de duplicidade
CREATE INDEX IF NOT EXISTS idx_ingress_dedup ON system.ingress_logs (project_slug, hook_id, event_id);

-- Comentário explicativo
COMMENT ON COLUMN system.ingress_logs.event_id IS 'ID único extraído do payload ou header para evitar processamento duplicado (Idempotency Key).';
