# Definição de Zonas de Rate Limit (Memória Compartilhada)
# Proteção contra DDoS e Força Bruta

# Zona Geral: 10MB de memória para rastrear IPs, limite de 50 requisições/segundo
limit_req_zone $binary_remote_addr zone=api_limit:10m rate=50r/s;

# Zona Estrita (Auth/Login): 10MB, limite de 5 requisições/segundo
limit_req_zone $binary_remote_addr zone=auth_limit:10m rate=5r/s;

# Zona de Conexões: Limita número de conexões simultâneas por IP
limit_conn_zone $binary_remote_addr zone=conn_limit:10m;

server {
    listen 80;
    server_name _;
    
    # Segurança: Ocultar versão do Nginx
    server_tokens off;

    # Segurança: Timeouts para evitar ataques Slowloris
    client_body_timeout 10s;
    client_header_timeout 10s;
    keepalive_timeout 5s;
    send_timeout 10s;

    # Segurança: Buffer Hardening (Previne Buffer Overflow)
    client_body_buffer_size 10K;
    client_header_buffer_size 1k;
    client_max_body_size 100M;
    large_client_header_buffers 2 1k;

    # Segurança: Headers de Proteção Básica
    add_header X-Frame-Options "SAMEORIGIN" always;
    add_header X-Content-Type-Options "nosniff" always;
    add_header X-XSS-Protection "1; mode=block" always;
    add_header Referrer-Policy "strict-origin-when-cross-origin" always;
    add_header Permissions-Policy "geolocation=(), microphone=(), camera=()" always;

    # Rota de Desafio do Certbot (Let's Encrypt)
    location /.well-known/acme-challenge/ {
        root /var/www/html;
        allow all;
    }

    # Frontend Management Studio
    location / {
        limit_req zone=api_limit burst=20 nodelay;
        limit_conn conn_limit 20;
        
        proxy_pass http://frontend;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }

    # Control Plane API
    location /api/control/ {
        limit_req zone=api_limit burst=10 nodelay;
        
        proxy_pass http://backend_control:3000;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }

    # Data Plane API
    location /api/data/ {
        limit_req zone=api_limit burst=50 nodelay;
        limit_conn conn_limit 50;

        proxy_pass http://backend_data:3000;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }
    
    # Auth Endpoints (Strict Limit)
    location ~ ^/api/data/.*/auth/(token|login|verify) {
        limit_req zone=auth_limit burst=5 nodelay;
        
        proxy_pass http://backend_data:3000;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }
    
    # Short URLs
    location ~ ^/(rpc|tables|auth|storage|realtime)/ {
        limit_req zone=api_limit burst=50 nodelay;
        
        proxy_pass http://backend_data:3000;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }
}
